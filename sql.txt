CREATE TABLE Breeds (
BreedID int NOT NULL AUTO_INCREMENT,
Description varchar(100),
PRIMARY KEY (BreedID)
);



CREATE TABLE Scores (
CompetitionID int,
ScoreID int,
Size int,
EyeColor int,
FurColor int
);

CREATE TABLE Dog (
DogID int NOT NULL AUTO_INCREMENT,
BreedID int,
DogName varchar(100),
DogGender varchar(5),
PRIMARY KEY (DogID),
FOREIGN KEY (BreedID) REFERENCES Breeds(BreedID)
);
CREATE TABLE Events_Table (
EventID int NOT NULL AUTO_INCREMENT,
Location varchar(100),
PRIMARY KEY (EventID)
);

CREATE TABLE Competitions (
CompetitionID int NOT NULL AUTO_INCREMENT,
EventID int,
Placement int,
DogID int,
ScoreID int,
PRIMARY KEY (CompetitionID),
FOREIGN KEY (EventID) REFERENCES Events_Table(EventID),
FOREIGN KEY (DogID) REFERENCES Dog(DogID),
FOREIGN KEY (ScoreID) REFERENCES Scores(ScoreID)
);
